#基于JQ的选项卡组件开发

```
<style>
    #div1 input, #div2 input, #div3 input{ cursor: pointer; }
    #div1 div, #div2 div, #div3 div{ width: 200px; height: 200px; border: 1px solid green; display: none; }
    .active{ background: #c40000; }
</style>

<script src="https://cdn.bootcss.com/jquery/1.10.2/jquery.min.js"></script>
<script>
    /**
     * title 基于JQ的选项卡组件
     * options event delay
     */

    $(function(){
        var t1 = new Tab();
        t1.init('div1', {});

        var t2 = new Tab();
        t2.init('div2', {
            event: 'mouseover'
        });

        var t3 = new Tab();
        t3.init('div3', {
            event: 'mouseover',
            delay: 200                  // 延迟200毫秒
        });
    });

    function Tab(){
        this.oParent = null;
        this.aInput = null;
        this.aDiv = null;
        this.settings = {               // 默认参数
            event: 'click',
            delay: 0
        };
    }

    Tab.prototype.init = function(oParent, opt){
        $.extend(this.settings, opt);

        this.oParent = $('#'+oParent);
        this.aInput = this.oParent.find('input');
        this.aDiv = this.oParent.find('div');

        this.change();
    };

    Tab.prototype.change = function(){
        var _this = this;
        var timer = null;
        this.aInput.on(this.settings.event, function () {
            if(_this.settings.event == 'mouseover' && _this.settings.delay){
                var obj = this;
                timer = setTimeout(function () {
                    show(obj);
                }, _this.settings.delay)
            }else{
                show(this);
            }
        }).on('mouseout', function(){
            clearTimeout(timer);
        });

        function show(obj){
            _this.aInput.attr('class', '');
            _this.aDiv.css('display', 'none');

            $(obj).attr('class', 'active');
            _this.aDiv.eq($(obj).index()).css('display', 'block');
        }
    };
</script>
```
